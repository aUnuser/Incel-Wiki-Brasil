{{/* https://www.bennettnotes.com/notes/hugo-responsive-images-with-markdown-render-hook/ */}} {{/* Original code from: https://laurakalbag.com/processing-responsive-images-with-hugo/ */}} {{/* Just modified a bit to work with render_image hook and output webp images */}} {{/* get file that matches the filename as specified as src="" */}} {{- $src := resources.Get .Destination -}} {{- $caption := .Text | safeHTML -}} {{- $class := .Title | safeHTML -}} {{- $format := "webp" -}} {{/* So for posts that aren't setup in the page bundles, it doesn't fail */}} {{- if and $src (ne (index (split .Destination ".") 1) "webm") -}} {{- $tinyw := default (printf "500x webp") -}} {{- $smallw := default (printf "800x webp") -}} {{- $mediumw := default (printf "1200x webp") -}} {{- $largew := default (printf "1500x webp") -}} {{/* resize the src image to the given sizes */}} {{/* We create a a temp scratch because it's not available in this context */}} {{- $data := newScratch }} {{- $data.Set "tiny" ($src.Resize $tinyw) -}} {{- $data.Set "small" ($src.Resize $smallw) -}} {{- $data.Set "medium" ($src.Resize $mediumw) -}} {{- $data.Set "large" ($src.Resize $largew) -}} {{/* add the processed images to the scratch */}} {{- $tiny := $data.Get "tiny" -}} {{- $small := $data.Get "small" -}} {{- $medium := $data.Get "medium" -}} {{- $large := $data.Get "large" -}} {{/* only use images smaller than or equal to the src (original) image size, as Hugo will upscale small images */}}
{{- $caption -}}
{{/* Since I do image-response class, the only thing that really matters is the height and width matches the image aspect ratio */}} {{- else if and $src (eq (index (split .Destination ".") 1) "webm") -}} {{ $noformat := replace $src ".webm" "" }} {{ $dimensions := split (index (split $noformat "-") 1) "x" }} {{ $width := index $dimensions 0 }} {{ $height := index $dimensions 1 }}
{{- $caption -}}
{{ end }}
